```js
function debounce(cb) {
  var timer = null;
  return function() {
    clearTimeout(timer);
    timer = setTimeout(() => {
      cb.apply(this, arguments);
    }, 1000)
  }
}
```

```js
function throttle(cb) {
  let canRun = true;
  return function() {
    if (!canRun) return;
    canRun = false;
    setTimeout(function() {
      cb.apply(this, arguments);
      canRun = true;
    }, 500);
  }
}
```
